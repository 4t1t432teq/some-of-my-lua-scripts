
local player = game.Players.LocalPlayer
local lastMoveTime = tick() 


local function randomJump()
 
    if math.random() < 0.005 then 
        local humanoid = player.Character:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.Jump = true  
        end
    end
end


local function moveRandomly()
   
    local function startMovement(character)
        local humanoid = character:WaitForChild("Humanoid")
        local rootPart = character:WaitForChild("HumanoidRootPart")

        while true do
         
            local randomDirection = Vector3.new(
                math.random(-50, 50),  
                0,                   
                math.random(-50, 50) 
            )

         
            local targetPosition = rootPart.Position + randomDirection

   
            humanoid:MoveTo(targetPosition)

          
            local distance = (rootPart.Position - targetPosition).Magnitude
            local walkingTime = distance / humanoid.WalkSpeed

         
            walkingTime = math.min(math.max(walkingTime, 5), 15) 

      
            local startTime = tick()
            while (rootPart.Position - targetPosition).Magnitude > 2 and (tick() - startTime) < walkingTime do
            
                randomJump()

                wait(0.5)  
            end

      
            wait(1) 
        end
    end

 
    player.CharacterAdded:Connect(function(character)
      
        character:WaitForChild("HumanoidRootPart")
        character:WaitForChild("Humanoid")
        
    
        startMovement(character)
    end)

 
    if player.Character then
        startMovement(player.Character)
    end
end


local function antiAFK()
    while true do
   
        if tick() - lastMoveTime >= 10 then 
            local character = player.Character
            if character then
                local humanoid = character:FindFirstChild("Humanoid")
                local rootPart = character:FindFirstChild("HumanoidRootPart")
                
                if humanoid and rootPart then
                 
                    rootPart.CFrame = rootPart.CFrame * CFrame.new(1, 0, 0) 

                  
                    lastMoveTime = tick()
                end
            end
        end
        wait(1)  
    end
end


moveRandomly()


antiAFK()
